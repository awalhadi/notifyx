Quick start

// Install
npm install notifyx

// Basic
import { NotifyX } from 'notifyx';
import 'notifyx/style.css';

NotifyX.success('Deployed!', { title: 'CI/CD' });

// AI streaming (OpenAI / Anthropic)
await NotifyX.stream(openai.chat.completions.stream({…}), {
  loadingMessage: 'AI is thinking…',
  onComplete: (text) => console.log(text),
});

// Promise toast
await NotifyX.promise(fetchData(), {
  loading: 'Fetching…',
  success: 'Done!',
  error: (e) => `Failed: ${e.message}`,
});

// With actions
NotifyX.error('Deployment failed.', {
  title: 'Build #491',
  actions: [
    { label: 'View Logs', onClick: () => openLogs() },
    { label: 'Retry', onClick: () => retry(), variant: 'primary' },
  ],
});
